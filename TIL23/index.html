Recursive Function

let a = {
  name: 'lee',
  b: {
    age: 99,
  },
};

function deepCopy(obj) {
  const result = {};
  for (let key in obj) {
    if (typeof obj[key] === 'object') {
      result[key] = deepCopy(obj[key]);
    } else {
      result[key] = obj[key];
    }
  }
  return result;
}

let newA = deepCopy(a);

newA.b.age = 100;
console.log('a:', a); //99
console.log('newA:', newA); //100
